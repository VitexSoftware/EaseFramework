 81:     {
 82:         if (is_array($parametersToChange) && count($parametersToChange)) {
 83:             foreach ($parametersToChange as $paramName => $paramValue) {
 84:                 if ($paramValue === true) {

 86:                 }
 87:             }
 88:             $targetParts = explode('&',

 90:             if (is_array($targetParts) && count($targetParts)) {
 91:                 $formTargetComputed = '';
 92:                 $targetPartsValues = [];
 93:                 foreach ($targetParts as $targetPart) {
 94:                     if (!strstr($targetPart, '=')) {
 95:                         $formTargetComputed .= $targetPart;
 96:                         continue;
 97:                     }
 98:                     list($targetPartName, $targetPartValue) = explode('=',
 99:                         $targetPart);
100:                     if ($targetPartValue === true) {
101:                         continue;
102:                     }
103:                     $targetPartsValues[$targetPartName] = $targetPartValue;
104:                 }
105:             }
106:             if ($replace) {
107:                 $newTargPartVals = array_merge($targetPartsValues,
108:                     $parametersToChange);
109:             } else {
110:                 $newTargPartVals = array_merge($parametersToChange,
111:                     $targetPartsValues);
112:             }
113:             $glueSign = '?';
114:             foreach ($newTargPartVals as $newTargetPartsValName => $newTargetPartsValue) {
115:                 $formTargetComputed .= $glueSign.urlencode($newTargetPartsValName).'='.urlencode($newTargetPartsValue);
116:                 $glueSign = '&';
117:             }
118:             $this->setFormTarget($formTargetComputed);
