 192:     public function xAttachObject($variableName, $ObjectName, $ObjectParams = null)
 193:     {
 194: 
 195:         if (!is_null($this->$variableName)) {
 196:             if (!is_object($this->$variableName)) {
 197:                 die('AttachObject: UndefinedProperty $this->' . $variableName . ' of ' . $this->getObjectName());
 198:             }
 199:         } else {
 200: 
 201:             if (class_exists($ObjectName)) {
 202: 
 203:                 if (property_exists($this, 'parentObject') && is_object($this->parentObject)) {
 204:                     //Pokud již rodičovský objekt obsahuje požadovanou vlastnost, pouze na ní odkážeme
 205:                     if (property_exists($this->parentObject, $variableName)) {
 206:                         $this->$variableName = & $this->parentObject->$variableName;
 207:                     }
 208:                 } else {
 209:                     //jinak se vytvoří nová instance objektu
 210:                     $NumberOfArgs = func_num_args();
 211:                     if ($NumberOfArgs == 2) {
 212:                         $this->$variableName = new $ObjectName();
 213:                     } else {
 214:                         $Arguments = func_get_args();
 215:                         array_shift($Arguments);
 216:                         array_shift($Arguments);
 217:                         eval('$this->' . $variableName . ' = new ' . $ObjectName . '(' . implode(',', $Arguments) . ');');
 218:                     }
 219:                 }
 220:             }
 221: 
 222:             return true;
 223:         }
 224: 
 225:         if (property_exists($this->$variableName, 'parentObject')) {
 226:             $this->$variableName->parentObject = & $this;

 803:     /**
 804:      * Oveření mailu
 805:      *
 806:      * @param string $email    mailová adresa
 807:      * @param bool   $checkDNS testovat DNS ?
 808:      *
 809:      * @package   isemail
 810:      * @author    Dominic Sayers <dominic_sayers@hotmail.com>
 811:      * @copyright 2009 Dominic Sayers
 812:      * @license   http://www.opensource.org/licenses/cpal_1.0 Common Public Attribution License Version 1.0 (CPAL) license
 813:      * @link      http://www.dominicsayers.com/isemail
 814:      * @version   1.9 - Minor modifications to make it compatible with PHPLint
 815:      */
 816:     static public function isEmail($email, $checkDNS = false)
 817:     {
 818:         /* Check that $email is a valid address. Read the following RFCs to understand the constraints:
 819:           //    (http://tools.ietf.org/html/rfc5322)
 820:           //    (http://tools.ietf.org/html/rfc3696)
 821:           //    (http://tools.ietf.org/html/rfc5321)
 822:           //    (http://tools.ietf.org/html/rfc4291#section-2.2)
 823:           //    (http://tools.ietf.org/html/rfc1123#section-2.1)
 824:           // the upper limit on address lengths should normally be considered to be 256
 825:           //    (http://www.rfc-editor.org/errata_search.php?rfc=3696)
 826:           //    NB I think John Klensin is misreading RFC 5321 and the the limit should actually be 254
 827:           //    However, I will stick to the published number until it is changed.
 828:           //
 829:           // The maximum total length of a reverse-path or forward-path is 256
 830:           // characters (including the punctuation and element separators)
 831:           //    (http://tools.ietf.org/html/rfc5321#section-4.5.3.1.3)
 832:          * */
 833:         $emailLength = strlen($email);
 834:         if ($emailLength > 256)
 835:             return false; // Too long
 836:             /*
 837:               // Contemporary email addresses consist of a "local part" separated from
 838:               // a "domain part" (a fully-qualified domain name) by an at-sign ("@").
 839:               //    (http://tools.ietf.org/html/rfc3696#section-3)
 840:              */
 841:         $atIndex = strrpos($email, '@');
 842: 
 843:         if ($atIndex === false) {
 844:             return false;
 845:         } // No at-sign
 846:         if ($atIndex === 0) {
 847:             return false;
 848:         } // No local part
 849:         if ($atIndex === $emailLength) {
 850:             return false;
 851:         } // No domain part
 852: // Sanitize comments
 853: // - remove nested comments, quotes and dots in comments
 854: // - remove parentheses and dots from quoted strings
 855:         $braceDepth = 0;
 856:         $inQuote = false;
 857:         $escapeThisChar = false;
 858: 
 859:         for ($i = 0; $i < $emailLength; ++$i) {
 860:             $char = $email[$i];
 861:             $replaceChar = false;
 862: 
 863:             if ($char === '\\') {
 864:                 $escapeThisChar = !$escapeThisChar; // Escape the next character?
 865:             } else {
 866:                 switch ($char) {
 867:                     case '(':
 868:                         if ($escapeThisChar) {
 869:                             $replaceChar = true;
 870:                         } else {
 871:                             if ($inQuote) {
 872:                                 $replaceChar = true;
 873:                             } else {
 874:                                 if ($braceDepth++ > 0) {
 875:                                     $replaceChar = true; // Increment brace depth
 876:                                 }
 877:                             }
 878:                         }
 879: 
 880:                         break;
 881:                     case ')':
 882:                         if ($escapeThisChar) {
 883:                             $replaceChar = true;
 884:                         } else {
 885:                             if ($inQuote) {
 886:                                 $replaceChar = true;
 887:                             } else {
 888:                                 if (--$braceDepth > 0) {
 889:                                     $replaceChar = true; // Decrement brace depth
 890:                                 }
 891:                                 if ($braceDepth < 0) {
 892:                                     $braceDepth = 0;
 893:                                 }
 894:                             }
 895:                         }
 896: 
 897:                         break;
 898:                     case '"':
 899:                         if ($escapeThisChar) {
 900:                             $replaceChar = true;
 901:                         } else {
 902:                             if ($braceDepth === 0) {
 903:                                 $inQuote = !$inQuote; // Are we inside a quoted string?
 904:                             } else {
 905:                                 $replaceChar = true;
 906:                             }
 907:                         }
 908: 
 909:                         break;
 910:                     case '.': // Dots don't help us either
 911:                         if ($escapeThisChar) {
 912:                             $replaceChar = true;
 913:                         } else {
 914:                             if ($braceDepth > 0)
 915:                                 $replaceChar = true;
 916:                         }
 917: 
 918:                         break;
 919:                     default:
 920:                 }
 921: 
 922:                 $escapeThisChar = false;
 923:                 if ($replaceChar) {
 924:                     $email[$i] = 'x';
 925:                 } // Replace the offending character with something harmless
 926:             }
 927:         }
 928: 
 929:         $localPart = substr($email, 0, $atIndex);
 930:         $domain = substr($email, $atIndex + 1);
 931:         $FWS = "(?:(?:(?:[ \\t]*(?:\\r\\n))?[ \\t]+)|(?:[ \\t]+(?:(?:\\r\\n)[ \\t]+)*))"; // Folding white space
 932: // Let's check the local part for RFC compliance...
 933: //
 934:         // local-part      =       dot-atom / quoted-string / obs-local-part
 935: // obs-local-part  =       word *("." word)
 936: //  (http://tools.ietf.org/html/rfc5322#section-3.4.1)
 937: //
 938:         // Problem: need to distinguish between "first.last" and "first"."last"
 939: // (i.e. one element or two). And I suck at regexes.
 940:         $dotArray = /* . (array[int]string) . */ preg_split('/\\.(?=(?:[^\\"]*\\"[^\\"]*\\")*(?![^\\"]*\\"))/m', $localPart);
 941:         $partLength = 0;
 942: 
 943:         foreach ($dotArray as $element) {
 944: // Remove any leading or trailing FWS
 945:             $element = preg_replace("/^$FWS|$FWS\$/", '', $element);
 946: 
 947: // Then we need to remove all valid comments (i.e. those at the start or end of the element
 948:             $elementLength = strlen($element);
 949: 
 950:             if ($element[0] === '(') {
 951:                 $indexBrace = strpos($element, ')');
 952:                 if ($indexBrace !== false) {
 953:                     if (preg_match('/(?<!\\\\)[\\(\\)]/', substr($element, 1, $indexBrace - 1)) > 0) {
 954:                         return false; // Illegal characters in comment
 955:                     }
 956:                     $element = substr($element, $indexBrace + 1, $elementLength - $indexBrace - 1);
 957:                     $elementLength = strlen($element);
 958:                 }
 959:             }
 960: 
 961:             if ($elementLength) {
 962:                 if ($element[$elementLength - 1] === ')') {
 963:                     $indexBrace = strrpos($element, '(');
 964:                     if ($indexBrace !== false) {
 965:                         if (preg_match('/(?<!\\\\)(?:[\\(\\)])/', substr($element, $indexBrace + 1, $elementLength - $indexBrace - 2)) > 0) {
 966:                             return false; // Illegal characters in comment
 967:                         }
 968:                         $element = substr($element, 0, $indexBrace);
 969:                         $elementLength = strlen($element);
 970:                     }
 971:                 }
 972:             }
 973: // Remove any leading or trailing FWS around the element (inside any comments)
 974:             $element = preg_replace("/^$FWS|$FWS\$/", '', $element);
 975: 
 976: // What's left counts towards the maximum length for this part
 977:             if ($partLength > 0) {
 978:                 $partLength++;
 979:             } // for the dot
 980:             $partLength += strlen($element);
 981: 
 982: // Each dot-delimited component can be an atom or a quoted string
 983: // (because of the obs-local-part provision)
 984:             if (preg_match('/^"(?:.)*"$/s', $element) > 0) {
 985: // Quoted-string tests:
 986: //
 987:                 // Remove any FWS
 988:                 $element = preg_replace("/(?<!\\\\)$FWS/", '', $element);
 989: // My regex skillz aren't up to distinguishing between \" \\" \\\" \\\\" etc.
 990: // So remove all \\ from the string first...
 991:                 $element = preg_replace('/\\\\\\\\/', ' ', $element);
 992:                 if (preg_match('/(?<!\\\\|^)["\\r\\n\\x00](?!$)|\\\\"$|""/', $element) > 0) {
 993:                     return false;
 994:                 } // ", CR, LF and NUL must be escaped, "" is too short
 995:             } else {
 996: // Unquoted string tests:
 997: //
 998:                 // Period (".") may...appear, but may not be used to start or end the
 999: // local part, nor may two or more consecutive periods appear.
1000: //  (http://tools.ietf.org/html/rfc3696#section-3)
1001: //
1002:                 // A zero-length element implies a period at the beginning or end of the
1003: // local part, or two periods together. Either way it's not allowed.
1004:                 if ($element === '') {
1005:                     return false;
1006:                 } // Dots in wrong place
1007: // Any ASCII graphic (printing) character other than the
1008: // at-sign ("@"), backslash, double quote, comma, or square brackets may
1009: // appear without quoting.  If any of that list of excluded characters
1010: // are to appear, they must be quoted
1011: //  (http://tools.ietf.org/html/rfc3696#section-3)
1012: //
1013:                 // Any excluded characters? i.e. 0x00-0x20, (, ), <, >, [, ], :, ;, @, \, comma, period, "
1014:                 if (preg_match('/[\\x00-\\x20\\(\\)<>\\[\\]:;@\\\\,\\."]/', $element) > 0) {
1015:                     return false;
1016:                 } // These characters must be in a quoted string
1017:             }
1018:         }
1019: 
1020:         if ($partLength > 64) {
1021:             return false;
1022:         } // Local part must be 64 characters or less
1023: // Now let's check the domain part...
1024: // The domain name can also be replaced by an IP address in square brackets
1025: //  (http://tools.ietf.org/html/rfc3696#section-3)
1026: //  (http://tools.ietf.org/html/rfc5321#section-4.1.3)
1027: //  (http://tools.ietf.org/html/rfc4291#section-2.2)
1028:         if (preg_match('/^\\[(.)+]$/', $domain) === 1) {
1029: // It's an address-literal
1030:             $addressLiteral = substr($domain, 1, strlen($domain) - 2);
1031:             $matchesIP = array();
1032: 
1033: // Extract IPv4 part from the end of the address-literal (if there is one)
1034:             if (preg_match('/\\b(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\\.) {3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/', $addressLiteral, $matchesIP) > 0) {
1035:                 $index = strrpos($addressLiteral, $matchesIP[0]);
1036: 
1037:                 if ($index === 0) {
1038: // Nothing there except a valid IPv4 address, so...
1039:                     return true;
1040:                 } else {
1041: // Assume it's an attempt at a mixed address (IPv6 + IPv4)
1042:                     if ($addressLiteral[$index - 1] !== ':') {
1043:                         return false;
1044:                     } // Character preceding IPv4 address must be ':'
1045:                     if (substr($addressLiteral, 0, 5) !== 'IPv6:') {
1046:                         return false;
1047:                     } // RFC5321 section 4.1.3
1048: 
1049:                     $IPv6 = substr($addressLiteral, 5, ($index === 7) ? 2 : $index - 6);
1050:                     $groupMax = 6;
1051:                 }
1052:             } else {
1053: // It must be an attempt at pure IPv6
1054:                 if (substr($addressLiteral, 0, 5) !== 'IPv6:') {
1055:                     return false;
1056:                 } // RFC5321 section 4.1.3
1057:                 $IPv6 = substr($addressLiteral, 5);
1058:                 $groupMax = 8;
1059:             }
1060: 
1061:             $groupCount = preg_match_all('/^[0-9a-fA-F]{0,4}|\\:[0-9a-fA-F]{0,4}|(.)/', $IPv6, $matchesIP);
1062:             $index = strpos($IPv6, '::');
1063: 
1064:             if ($index === false) {
1065: // We need exactly the right number of groups
1066:                 if ($groupCount !== $groupMax) {
1067:                     return false;
1068:                 } // RFC5321 section 4.1.3
1069:             } else {
1070:                 if ($index !== strrpos($IPv6, '::')) {
1071:                     return false;
1072:                 } // More than one '::'
1073:                 $groupMax = ($index === 0 || $index === ( strlen($IPv6) - 2)) ? $groupMax : $groupMax - 1;
1074:                 if ($groupCount > $groupMax) {
1075:                     return false;
1076:                 } // Too many IPv6 groups in address
1077:             }
1078: 
1079: // Check for unmatched characters
1080:             array_multisort($matchesIP[1], SORT_DESC);
1081:             if ($matchesIP[1][0] !== '') {
1082:                 return false;
1083:             } // Illegal characters in address
1084: // It's a valid IPv6 address, so...
1085:             return true;
1086:         } else {
1087: // It's a domain name...
1088: // The syntax of a legal Internet host name was specified in RFC-952
1089: // One aspect of host name syntax is hereby changed: the
1090: // restriction on the first character is relaxed to allow either a
1091: // letter or a digit.
1092: //  (http://tools.ietf.org/html/rfc1123#section-2.1)
1093: //
1094:             // NB RFC 1123 updates RFC 1035, but this is not currently apparent from reading RFC 1035.
1095: //
1096:             // Most common applications, including email and the Web, will generally not
1097: // permit...escaped strings
1098: //  (http://tools.ietf.org/html/rfc3696#section-2)
1099: //
1100:             // the better strategy has now become to make the "at least one period" test,
1101: // to verify LDH conformance (including verification that the apparent TLD name
1102: // is not all-numeric)
1103: //  (http://tools.ietf.org/html/rfc3696#section-2)
1104: //
1105:             // Characters outside the set of alphabetic characters, digits, and hyphen MUST NOT appear in domain name
1106: // labels for SMTP clients or servers
1107: //  (http://tools.ietf.org/html/rfc5321#section-4.1.2)
1108: //
1109:             // RFC5321 precludes the use of a trailing dot in a domain name for SMTP purposes
1110: //  (http://tools.ietf.org/html/rfc5321#section-4.1.2)
1111:             $dotArray = /* . (array[int]string) . */ preg_split('/\\.(?=(?:[^\\"]*\\"[^\\"]*\\")*(?![^\\"]*\\"))/m', $domain);
1112:             $partLength = 0;
1113: 
1114:             if (count($dotArray) === 1) {
1115:                 return false;
1116:             } // Mail host can't be a TLD
1117: 
1118:             foreach ($dotArray as $element) {
1119: // Remove any leading or trailing FWS
1120:                 $element = preg_replace("/^$FWS|$FWS\$/", '', $element);
1121: 
1122: // Then we need to remove all valid comments (i.e. those at the start or end of the element
1123:                 $elementLength = strlen($element);
1124: 
1125:                 if ($element[0] === '(') {
1126:                     $indexBrace = strpos($element, ')');
1127:                     if ($indexBrace !== false) {
1128:                         if (preg_match('/(?<!\\\\)[\\(\\)]/', substr($element, 1, $indexBrace - 1)) > 0) {
1129:                             return false; // Illegal characters in comment
1130:                         }
1131:                         $element = substr($element, $indexBrace + 1, $elementLength - $indexBrace - 1);
1132:                         $elementLength = strlen($element);
1133:                     }
1134:                 }
1135: 
1136:                 if ($element[$elementLength - 1] === ')') {
1137:                     $indexBrace = strrpos($element, '(');
1138:                     if ($indexBrace !== false) {
