  72:      * Projde $this->raiseItems (metoda_potomka=>proměnná_rodiče) a pokud v
  73:      * objektu najde metodu potomka, zavolá jí s parametrem
  74:      * $this->proměnná_rodiče
  75:      *
  76:      * @param object $childObject  vkládaný objekt
  77:      * @param array  $itemsToRaise pole položek k "protlačení"
  78:      */
  79:     public function raise(& $childObject, $itemsToRaise = null)
  80:     {
  81:         if (!$itemsToRaise) {
  82:             $itemsToRaise = $childObject->raiseItems;
  83:         }
  84: 
  85:         foreach ($itemsToRaise as $method => $property) {
  86:             if (method_exists($childObject, $method)) {
  87:                 if (isset($this->$property)) {
  88:                     $childObject->$method($this->$property);
  89:                 }
  90:             } else {
  91:                 if (isset($this->$property)) {
  92:                     $childObject->$property = & $this->$property;
  93:                 }
  94:             }
  95:         }
  96:     }

  98:     /**
  99:      * Vloží další element do objektu
 100:      *
 101:      * @param mixed  $pageItem     hodnota nebo EaseObjekt s metodou draw()
 102:      * @param mixed  $context      Objekt do nějž jsou prvky/položky vkládány
 103:      * @param string $pageItemName Pod tímto jménem je objekt vkládán do stromu
 104:      *
 105:      * @return pointer Odkaz na vložený objekt
 106:      */
 107:     static function &addItemCustom($pageItem, $context, $pageItemName = null)
 108:     {
 109:         $itemPointer = null;
 110:         if (is_object($pageItem)) {
 111:             if (method_exists($pageItem, 'draw')) {
 112:                 $duplicity = 1;
 113:                 if (is_null($pageItemName) || !strlen($pageItemName)) {
 114:                     $pageItemName = $pageItem->getObjectName();
 115:                 }
 116: 
 117:                 while (isset($context->pageParts[$pageItemName])) {
 118:                     $pageItemName = $pageItemName . $duplicity++;
 119:                 }
 120: 
 121:                 $context->pageParts[$pageItemName] = $pageItem;
 122:                 $context->pageParts[$pageItemName]->parentObject = & $context;
 123: 
 124:                 if (
 125:                     isset($context->pageParts[$pageItemName]->raiseItems) &&
 126:                     is_array($context->pageParts[$pageItemName]->raiseItems) &&
 127:                     count($context->pageParts[$pageItemName]->raiseItems)
 128:                 ) {
 129:                     $context->raise($context->pageParts[$pageItemName]);
 130:                 }
 131:                 if (method_exists($context->pageParts[$pageItemName], 'AfterAdd')) {
 132:                     $context->pageParts[$pageItemName]->afterAdd();
 133:                 }
 134:                 $context->lastItem = & $context->pageParts[$pageItemName];
 135:                 $itemPointer = & $context->pageParts[$pageItemName];
 136:             } else {
 137:                 $context->error('Page Item object without draw() method', $pageItem);
 138:             }
 139:         } else {
 140:             if (is_array($pageItem)) {
 141:                 $addedItemPointer = $context->addItems($pageItem);
 142:                 $itemPointer = & $addedItemPointer;
 143:             } else {
 144:                 if (!is_null($pageItem)) {
 145:                     $context->pageParts[] = $pageItem;
 146:                     $EndPointer = end($context->pageParts);
 147:                     $itemPointer = &$EndPointer;
 148:                 }
 149:             }
 150:         }
 151:         EaseShared::instanced()->registerItem($itemPointer);
 152: 
 153:         return $itemPointer;
 154:     }
