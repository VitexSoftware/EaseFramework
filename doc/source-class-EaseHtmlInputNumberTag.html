 467:      * @param array  $itemValues
 468:      * @param array  $tagProperties
 469:      */
 470:     public function __construct($name, $items = null, $itemValues = null, $tagProperties = null)
 471:     {
 472:         parent::__construct($name, $items, $tagProperties);
 473:         if (!is_null($itemValues)) {
 474:             $values = array();
 475:             foreach ($itemValues as $itemName => $item) {
 476:                 $values[$name . '_' . $itemName] = $item;
 477:             }
 478:             $this->setValues($values);
 479:         }
 480:     }
 481: 
 482:     /**
 483:      * Přejmenuje vložené checkboxy pro použití ve formuláři
 484:      *
 485:      * @param EaseHtmlCheckboxTag $pageItem     vkládaný objekt CheckBoxu
 486:      * @param string              $pageItemName Pod tímto jménem je objekt vkládán do stromu
 487:      *
 488:      * @return EaseHtmlCheckboxTag
 489:      */
 490:     function &addItem($pageItem, $pageItemName = null)
 491:     {
 492:         /**
 493:          * Allready Added Item
 494:          * @var EaseHtmlCheckboxTag
 495:          */
 496:         $itemInpage = parent::addItem($pageItem);
 497:         if (is_object($itemInpage)) {
 498:             if (isset($this->items)) {
 499:                 $keys = array_keys($this->items);
 500:                 $itemInpage->setTagProperties(array('name' => $itemInpage->getTagProperty('name') . '#' . $keys[$this->_subitemCount]));
 501:                 if (isset($this->values[$keys[$this->_subitemCount]])) {
 502:                     $itemInpage->setValue((bool) $this->values[$keys[$this->_subitemCount]]);
 503:                 }
 504:                 next($this->items);
 505:                 $this->_subitemCount++;
 506:             }
 507:         }
 508: 
 509:         return $itemInpage;
 510:     }
 511: 
 512:     /**

 514:      */
 515:     public function finalize()
 516:     {
 517:         parent::finalize();

 519:     }
 520:
